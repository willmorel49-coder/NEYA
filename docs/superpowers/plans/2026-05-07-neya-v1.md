```jsx
// Fade : monte ou descend l'opacité d'un enfant
// in=true → fondu entrant, in=false → fondu sortant
function Fade({ children, duration = 800, className = '' }) {
  const [visible, setVisible] = useState(false)

  useEffect(() => {
    const t = requestAnimationFrame(() => setVisible(true))
    return () => cancelAnimationFrame(t)
  }, [])

  return (
    <div
      className={className}
      style={{
        opacity: visible ? 1 : 0,
        transition: `opacity ${duration}ms ease`,
      }}
    >
      {children}
    </div>
  )
}
```

```jsx
// ─── AUDIO ───────────────────────────────────────────────────────────────────

function createAudioContext() {
  return new (window.AudioContext || window.webkitAudioContext)()
}

// Génère un buffer de bruit blanc
function createNoiseBuffer(ctx, seconds = 3) {
  const len = ctx.sampleRate * seconds
  const buf = ctx.createBuffer(1, len, ctx.sampleRate)
  const data = buf.getChannelData(0)
  for (let i = 0; i < len; i++) data[i] = Math.random() * 2 - 1
  return buf
}

// Lance une ambiance sonore et retourne une fonction stop()
function startAmbience(type, volume = 0.08) {
  if (type === 'silence') return () => {}

  const ctx = createAudioContext()
  const gainNode = ctx.createGain()
  gainNode.gain.setValueAtTime(0, ctx.currentTime)
  gainNode.gain.linearRampToValueAtTime(volume, ctx.currentTime + 1.5)
  gainNode.connect(ctx.destination)

  const buf = createNoiseBuffer(ctx, 4)
  const source = ctx.createBufferSource()
  source.buffer = buf
  source.loop = true

  if (type === 'pluie') {
    // Bruit blanc filtré haut — texture pluie
    const filter = ctx.createBiquadFilter()
    filter.type = 'highpass'
    filter.frequency.value = 2000
    source.connect(filter)
    filter.connect(gainNode)
  } else if (type === 'vent') {
    // Bruit filtré passe-bande — texture vent
    const filter = ctx.createBiquadFilter()
    filter.type = 'bandpass'
    filter.frequency.value = 400
    filter.Q.value = 0.5
    source.connect(filter)
    filter.connect(gainNode)
  } else if (type === 'feu') {
    // Bruit brun (grave) — texture feu
    const filter = ctx.createBiquadFilter()
    filter.type = 'lowpass'
    filter.frequency.value = 300
    source.connect(filter)
    filter.connect(gainNode)
  }

  source.start()

  return () => {
    gainNode.gain.linearRampToValueAtTime(0, ctx.currentTime + 0.8)
    setTimeout(() => ctx.close(), 1000)
  }
}
```

```jsx
function WorldReveal({ ritual, world, onGoVrai, muted }) {

  const [displayedPhrase, setDisplayedPhrase] = useState('')
  const fullPhrase = selectPhrase(ritual, world)
  const stopSoundRef = useRef(() => {})

  useEffect(() => {
    // Phase 1 : noir (800ms)
    const t1 = setTimeout(() => {
      setPhase('world')
      // Relancer le son choisi en fond très bas
      if (!muted && ritual.sound && ritual.sound !== 'silence') {
        stopSoundRef.current = startAmbience(ritual.sound, 0.04)
      }
    }, 800)

    // Phase 2 : phrase (après 2000ms dans le monde)
    const t2 = setTimeout(() => setPhase('phrase'), 2800)

    return () => { clearTimeout(t1); clearTimeout(t2); stopSoundRef.current() }
  }, []) // eslint-disable-line

  // Phrase lettre par lettre
  useEffect(() => {
    if (phase !== 'phrase') return
    let i = 0

      i++
```
